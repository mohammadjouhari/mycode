$(document).ready(function () {
    $('#adminTree').jstree({
        'core': {
            'check_callback': true,
            'data': [
                { "id": "1", "parent": "#", "text": "Example Root" }
            ]
        },
        'plugins': ["wholerow", "contextmenu"] 
    });

    $('#adminTree').on("select_node.jstree", function (e, data) {
        $('#nodeId').val(data.node.id);
        $('#nodeText').val(data.node.text);
    });
});

// Function to add a ROOT node
function createNewNode() {
    var tree = $('#adminTree').jstree(true);
    var newNodeId = tree.create_node("#", { "text": "New Root" }, "last");
    if(newNodeId) tree.edit(newNodeId);
}

// Function to add a CHILD node to the currently selected node
function createChildNode() {
    var tree = $('#adminTree').jstree(true);
    var sel = tree.get_selected();
    if(!sel.length) { alert("Please select a parent node first"); return; }
    
    var newNodeId = tree.create_node(sel[0], { "text": "New Child" }, "last");
    if(newNodeId) {
        tree.open_node(sel[0]);
        tree.edit(newNodeId);
    }
}

function saveNode() {
    var tree = $('#adminTree').jstree(true);
    var id = $('#nodeId').val();
    var newText = $('#nodeText').val();

    if (id) {
        tree.rename_node(id, newText);
        
        // 1. Get raw flat JSON
        var rawData = tree.get_json('#', { flat: true });

        // 2. Clean the JSON to have ONLY id, text, and parent
        var cleanJson = rawData.map(function(node) {
            return {
                id: node.id,
                text: node.text,
                parent: node.parent
            };
        });

        console.log("Clean JSON for Backend:", cleanJson);
        // This cleanJson is what you send to your .NET Controller
    }
}
