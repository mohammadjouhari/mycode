<script>
   $(document).ready(function() {
    // 1. Initialize and assign to a variable
    var table = $('#sessionTable').DataTable({
        responsive: true,
        pageLength: 5,
        language: {
            search: "_INPUT_",
            searchPlaceholder: "Search sessions...",
        },
        order: [[3, 'desc']],
        columnDefs: [
            { orderable: false, targets: 5 } 
        ]
    });

    // 2. The handler MUST be inside here to "see" the 'table' variable
    $('#applyFilters').on('click', function() {
        var categoryValue = $('#categoryFilter').val();
        var dateValue = $('#dateFilter').val();

        // If the user selects "All Sessions", clear the category search
        if (categoryValue === "All Sessions") {
            table.column(2).search('');
        } else {
            table.column(2).search(categoryValue);
        }

        // Apply Date Filter (Column index 3)
        table.column(3).search(dateValue);

        // Finalize and refresh the view
        table.draw();
    });
});
</script>
