const tagInput = document.getElementById('tagInput');
const wrapper = document.getElementById('tagInputWrapper');
const hiddenInput = document.getElementById('hiddenTags');
let tags = [];

tagInput.addEventListener('keydown', function(e) {
    if (e.key === 'Enter') {
        e.preventDefault(); // Prevent form submission
        const value = tagInput.value.trim();
        
        if (value && !tags.includes(value)) {
            tags.push(value);
            addTagToUI(value);
            tagInput.value = '';
            updateHiddenInput();
        }
    }
});

function addTagToUI(text) {
    const span = document.createElement('span');
    span.className = 'badge bg-primary tag-badge p-2 rounded-pill';
    span.innerHTML = `${text} <button type="button" class="btn-close btn-close-white" aria-label="Remove"></button>`;
    
    // Remove tag on click
    span.querySelector('.btn-close').addEventListener('click', () => {
        tags = tags.filter(t => t !== text);
        span.remove();
        updateHiddenInput();
    });

    wrapper.insertBefore(span, tagInput);
}

function updateHiddenInput() {
    hiddenInput.value = tags.join(','); // Sends "Tag1,Tag2,Tag3" to .NET
}

const myButton = document.getElementById('btnSubmit');

myButton.addEventListener('click', function(e) {
    alert(tags);
});
