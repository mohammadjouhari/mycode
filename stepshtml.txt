<script>
    const formSteps = document.querySelectorAll(".form-step");
    const stepperItems = document.querySelectorAll(".stepper-item");
    const nextBtn = document.getElementById("nextBtn");
    const prevBtn = document.getElementById("prevBtn");
    const form = document.getElementById("multiStepForm");
    
    let currentStep = 0;

    function updateStepView() {
        // Toggle active visual states on form panel views
        formSteps.forEach((step, index) => {
            step.classList.toggle("active", index === currentStep);
        });

        // Sync visual completion UI states on your provided stepper items
        stepperItems.forEach((item, index) => {
            item.classList.toggle("active", index === currentStep);
            item.classList.toggle("completed", index < currentStep);
        });

        // Adjust back/forward buttons text configurations
        prevBtn.disabled = currentStep === 0;
        
        if (currentStep === formSteps.length - 1) {
            nextBtn.textContent = "Submit Form";
            nextBtn.classList.replace("btn-primary", "btn-success");
        } else {
            nextBtn.textContent = "Next Step";
            nextBtn.classList.replace("btn-success", "btn-primary");
        }
    }

    // Custom helper engine checking constraints inside current panel space only
    function isCurrentStepValid() {
        const activeFields = formSteps[currentStep].querySelectorAll("input, select, textarea");
        let valid = true;
        
        activeFields.forEach(field => {
            if (!field.checkValidity()) {
                valid = false;
                field.classList.add("is-invalid");
            } else {
                field.classList.remove("is-invalid");
            }
        });
        return valid;
    }

    nextBtn.addEventListener("click", () => {
        if (!isCurrentStepValid()) return; // Stop user from progressing if inputs are incomplete

        if (currentStep < formSteps.length - 1) {
            currentStep++;
            updateStepView();
        } else {
            // Reaching step 5 triggers actual processing
            alert("Form completely submitted! Sending tracking data to server payload...");
            form.submit();
        }
    });

    prevBtn.addEventListener("click", () => {
        if (currentStep > 0) {
            currentStep--;
            updateStepView();
        }
    });
</script>
