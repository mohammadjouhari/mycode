Since you are already comfortable with Bootstrap and jQuery, integrating jsTree is a great way to handle hierarchical data—like a file explorer for your LibraryManagement project or a category tree for your E-Shop.

Here is a complete, single-file sample. It uses a Mock JSON structure and the default Bootstrap-compatible styling.

1. The Complete HTML/JS Sample

    <link rel="stylesheet" href="https://cdnjs.cloudflare.com/ajax/libs/jstree/3.3.12/themes/default/style.min.css" />
    
    <style>

<script src="https://cdnjs.cloudflare.com/ajax/libs/jquery/3.4.1/jquery.min.js"></script>
<script src="https://cdnjs.cloudflare.com/ajax/libs/jstree/3.3.12/jstree.min.js"></script>

<script>
    $(document).ready(function () {
        // 1. Sample JSON Data structure
        const treeData = [
            { "id": "ajson1", "parent": "#", "text": "E-Shop Products", "state": { "opened": true } },
            { "id": "ajson2", "parent": "ajson1", "text": "Electronics" },
            { "id": "ajson3", "parent": "ajson1", "text": "Fashion" },
            { "id": "ajson4", "parent": "ajson2", "text": "Laptops", "icon": "glyphicon glyphicon-leaf" },
            { "id": "ajson5", "parent": "ajson2", "text": "Smartphones" },
            { "id": "ajson6", "parent": "ajson3", "text": "Menswear" },
            { "id": "ajson7", "parent": "ajson3", "text": "Accessories" }
        ];

        // 2. Initialize jsTree
        $('#data-tree').jstree({
            'core': {
                'data': treeData,
                'check_callback': true, // allows us to modify the tree
                'themes': {
                    'name': 'default',
                    'responsive': true
                }
            },
            'plugins': ['search', 'types', 'wholerow']
        });

        // 3. Handle selection events
        $('#data-tree').on("changed.jstree", function (e, data) {
            if(data.selected.length) {
                const node = data.instance.get_node(data.selected[0]);
                $('#selection-text').html(`Selected: <strong>${node.text}</strong> (ID: ${node.id})`);
                logEvent(`User selected: ${node.text}`);
            }
        });

        // 4. Manual trigger
        $('#get-selected-btn').click(function() {
            const selectedNode = $('#data-tree').jstree('get_selected');
            alert("Selected ID: " + selectedNode);
        });

        function logEvent(msg) {
            const timestamp = new Date().toLocaleTimeString();
            $('#event-log').prepend(`[${timestamp}] ${msg}<br>`);
        }
    });
</script>

</body>
