import { Injectable } from '@angular/core';
import { CanActivate, Router } from '@angular/router';
import { JwtHelperService } from '@auth0/angular-jwt';

@Injectable({
  providedIn: 'root'
})
export class AuthGuard implements CanActivate {
  private jwtHelper = new JwtHelperService();

  constructor(private router: Router) {}

  canActivate(): boolean {
    const token = localStorage.getItem('token');

    if (token) {
      // 1. Validate Expiration (exp)
      if (this.jwtHelper.isTokenExpired(token)) {
        this.router.navigate(['login']);
        return false;
      }

      // Decode the token to check Issuer and Audience
      const decodedToken = this.jwtHelper.decodeToken(token);

      // 2. Validate Issuer (iss)
      const expectedIssuer = 'https://your-dotnet-api.com';
      if (decodedToken.iss !== expectedIssuer) {
        console.error('Invalid Issuer');
        return false;
      }

      // 3. Validate Audience (aud)
      const expectedAudience = 'ecommerce-app';
      if (decodedToken.aud !== expectedAudience) {
        console.error('Invalid Audience');
        return false;
      }

      return true; // Everything passed
    }

    this.router.navigate(['login']);
    return false;
  }
}
