       function changeImage(src) {
    // 1. Get the main image element by its ID
    const mainImg = document.getElementById('mainProductImg');
    
    // 2. Update the main image source
    mainImg.src = src;

    // 3. Optional: Update the 'active' class for the thumbnails
    // Remove 'active' from all thumbnails
    const thumbnails = document.querySelectorAll('.thumb-img');
    thumbnails.forEach(img => img.classList.remove('active'));

    // Find the thumbnail that was clicked and add 'active'
    // (Note: This works best if you pass the element 'this' instead of just 'this.src')
    const clickedThumb = event.target;
    clickedThumb.classList.add('active');
       }
     </script>
