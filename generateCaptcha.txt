let currentCaptcha = "";

function generateCaptcha() {
    const canvas = document.getElementById("captchaCanvas");
    const ctx = canvas.getContext("2d");
    const characters = "ABCDEFGHJKLMNPQRSTUVWXYZabcdefghijkmnpqrstuvwxyz23456789";
    currentCaptcha = "";
    
    // Clear canvas
    ctx.clearRect(0, 0, canvas.width, canvas.height);

    // 1. Generate Random String (6 chars)
    for (let i = 0; i < 6; i++) {
        currentCaptcha += characters.charAt(Math.floor(Math.random() * characters.length));
    }

    // 2. Styling the text
    ctx.font = "bold 28px 'Courier New'";
    ctx.textBaseline = "middle";
    ctx.fillStyle = "#0056b3"; // Your brand blue

    // 3. Draw text with slight rotation/randomness
    for (let i = 0; i < currentCaptcha.length; i++) {
        const x = 20 + (i * 20);
        const y = 25 + (Math.random() * 10 - 5);
        ctx.save();
        ctx.translate(x, y);
        ctx.rotate((Math.random() * 10 - 5) * Math.PI / 180);
        ctx.fillText(currentCaptcha[i], 0, 0);
        ctx.restore();
    }

    // 4. Add "Noise" (Random lines to confuse bots)
    for (let i = 0; i < 5; i++) {
        ctx.strokeStyle = "#ccc";
        ctx.beginPath();
        ctx.moveTo(Math.random() * canvas.width, Math.random() * canvas.height);
        ctx.lineTo(Math.random() * canvas.width, Math.random() * canvas.height);
        ctx.stroke();
    }
}

function validateCaptcha() {
    const userInput = document.getElementById("userInput").value;
    const feedback = document.getElementById("feedback");

    if (userInput === currentCaptcha) {
        feedback.innerHTML = '<span class="text-success">✔ Correct! Form submitted.</span>';
        // Here you would call your .NET API: _service.CreateSomething(request);
    } else {
        feedback.innerHTML = '<span class="text-danger">✖ Incorrect code. Try again.</span>';
        generateCaptcha(); // Force a new code on failure
    }
}

// Initial call
window.onload = generateCaptcha;
